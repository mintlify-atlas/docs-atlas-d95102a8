Get started with Axmed Design System and build your first component in minutes.

<Steps>
  <Step title="Install the package">
    Add Axmed Design System to your React project:

    <CodeGroup>
    ```bash npm
    npm install axmed-design-system antd @ant-design/icons react react-dom
    ```

    ```bash yarn
    yarn add axmed-design-system antd @ant-design/icons react react-dom
    ```

    ```bash pnpm
    pnpm add axmed-design-system antd @ant-design/icons react react-dom
    ```
    </CodeGroup>

    <Note>
      Axmed Design System requires React 18+, Ant Design 5+, and @ant-design/icons 5+ as peer dependencies.
    </Note>
  </Step>

  <Step title="Import the CSS">
    Import the design system's stylesheet in your application entry point (e.g., `main.tsx` or `App.tsx`):

    ```tsx
    import 'axmed-design-system/style.css'
    ```

    This imports all design tokens (typography, colors, spacing) and component styles. The Figtree font is loaded automatically.
  </Step>

  <Step title="Build your first component">
    Create a simple welcome card using Axmed components:

    ```tsx
    import { AxButton, AxCard, AxText } from 'axmed-design-system'

    function WelcomeCard() {
      return (
        <AxCard
          title="Welcome to Axmed"
          description="Healthcare marketplace platform"
        >
          <AxText variant="body-md" color="secondary">
            Get started by exploring our component library and building
            production-ready healthcare applications.
          </AxText>
          
          <div style={{ marginTop: '16px', display: 'flex', gap: '8px' }}>
            <AxButton variant="primary">
              Get Started
            </AxButton>
            <AxButton variant="secondary">
              View Docs
            </AxButton>
          </div>
        </AxCard>
      )
    }

    export default WelcomeCard
    ```
  </Step>

  <Step title="Explore core components">
    Try out the essential components:

    <Accordion title="AxButton - All variants">
      ```tsx
      import { AxButton } from 'axmed-design-system'
      import { PlusOutlined } from '@ant-design/icons'

      function ButtonShowcase() {
        return (
          <div style={{ display: 'flex', gap: '8px', flexWrap: 'wrap' }}>
            <AxButton variant="primary">Primary</AxButton>
            <AxButton variant="secondary">Secondary</AxButton>
            <AxButton variant="ghost">Ghost</AxButton>
            <AxButton variant="danger">Danger</AxButton>
            <AxButton variant="primary" icon={<PlusOutlined />}>
              With Icon
            </AxButton>
          </div>
        )
      }
      ```
    </Accordion>

    <Accordion title="AxText - Typography variants">
      ```tsx
      import { AxText } from 'axmed-design-system'

      function TypographyShowcase() {
        return (
          <div>
            <AxText variant="heading-2xl">Extra Large Heading</AxText>
            <AxText variant="heading-lg" weight="semibold">
              Section Title
            </AxText>
            <AxText variant="body-md" color="secondary">
              This is body text with secondary color
            </AxText>
            <AxText variant="body-sm" color="link">
              Learn more →
            </AxText>
          </div>
        )
      }
      ```
    </Accordion>

    <Accordion title="AxCard - Content containers">
      ```tsx
      import { AxCard, AxText, AxButton } from 'axmed-design-system'

      function OrderCard() {
        return (
          <AxCard
            title="Order #ORD-2024-1234"
            description="Pharmaceutical supplies order"
            extra={<AxButton variant="text" size="small">View Details</AxButton>}
            footer={
              <div style={{ display: 'flex', justifyContent: 'space-between' }}>
                <AxText variant="body-sm" color="secondary">
                  Due: Jan 15, 2024
                </AxText>
                <AxText variant="body-sm" weight="semibold">
                  $12,450.00
                </AxText>
              </div>
            }
          >
            <AxText variant="body-md">
              45 items • Supplier: MedPharm Inc.
            </AxText>
          </AxCard>
        )
      }
      ```
    </Accordion>
  </Step>
</Steps>
